安装测试依赖

```
npm i -D mocha chai
```


配置npm scripts

```
 "scripts": {
    "test": "./node_modules/.bin/mocha -u bdd"
  },
```

创建第一个测试test/test.js

```
var assert = require('assert');
describe('Array', function() {
  describe('#indexOf()', function() {
    it('should return -1 when the value is not present', function() {
      assert.equal(-1, [1,2,3].indexOf(4));
    });
  });
});
```

执行npm test
